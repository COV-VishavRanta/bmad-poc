### Assignment Form Design and Validation
[Source: docs/prd/epic-7-team-assignment.md]
Assignment form with real-time validation:
```typescript
interface AssignmentFormData {
  user_id: number | null
  project_id: number | null
  role_name: string
  fte_allocation: number
  start_date: string
  end_date: string
  notes: string
}

interface AssignmentFormValidation {
  user_id: ValidationResult
  project_id: ValidationResult
  role_name: ValidationResult
  fte_allocation: ValidationResult
  start_date: ValidationResult
  end_date: ValidationResult
  capacity_impact: CapacityValidationResult
  assignment_conflicts: AssignmentConflict[]
}

const useAssignmentForm = () => {
  const [formData, setFormData] = useState<AssignmentFormData>(initialState)
  const [validation, setValidation] = useState<AssignmentFormValidation>({})
  const [conflicts, setConflicts] = useState<AssignmentConflict[]>([])
  
  const validateFormField = useCallback(async (field: string, value: any) => {
    // Real-time field validation with conflict checking
    const fieldValidation = await validateAssignmentField(field, value, formData)
    setValidation(prev => ({ ...prev, [field]: fieldValidation }))
    
    // Check for assignment conflicts when key fields change
    if (['user_id', 'project_id', 'start_date', 'end_date', 'fte_allocation'].includes(field)) {
      const potentialAssignment = { ...formData, [field]: value }
      const detectedConflicts = await detectAssignmentConflicts(potentialAssignment)
      setConflicts(detectedConflicts)
    }
  }, [formData])
  
  const submitForm = useCallback(async () => {
    const finalValidation = await validateCompleteAssignment(formData)
    if (finalValidation.isValid) {
      await createAssignment(formData)
    } else {
      setValidation(finalValidation.fieldValidations)
      setConflicts(finalValidation.conflicts)
    }
  }, [formData])
  
  return { formData, setFormData, validation, conflicts, validateFormField, submitForm }
}
```

interface ConflictDetectionProps {
  assignment: Assignment
  onConflictResolution: (resolution: ConflictResolution) => void
}

const AssignmentConflicts: React.FC<ConflictDetectionProps> = ({
  assignment,
  onConflictResolution
}) => {
  const [conflicts, setConflicts] = useState<AssignmentConflict[]>([])
  const [selectedResolution, setSelectedResolution] = useState<ConflictResolution | null>(null)
  
  useEffect(() => {
    detectAssignmentConflicts(assignment).then(setConflicts)
  }, [assignment])
  
  const handleConflictResolution = async (conflict: AssignmentConflict, resolution: ConflictResolution) => {
    const result = await resolveAssignmentConflict(conflict.id, resolution)
    if (result.success) {
      setConflicts(prev => prev.filter(c => c.id !== conflict.id))
      onConflictResolution(resolution)
    }
  }
  
  return (
    <div className="assignment-conflicts">
      {conflicts.length > 0 && (
        <Alert severity="warning" className="conflict-alert">
          <AlertTitle>Assignment Conflicts Detected</AlertTitle>
          {conflicts.length} conflict{conflicts.length > 1 ? 's' : ''} found that require resolution
        </Alert>
      )}
      
      {conflicts.map(conflict => (
        <ConflictCard 
          key={conflict.id}
          conflict={conflict}
          onResolve={(resolution) => handleConflictResolution(conflict, resolution)}
        />
      ))}
      
      {conflicts.length > 0 && (
        <ConflictResolutionWizard 
          conflicts={conflicts}
          onBatchResolve={handleBatchResolution}
        />
      )}
    </div>
  )
}

interface AssignmentCalendarProps {

const AssignmentCalendar: React.FC<AssignmentCalendarProps> = ({

  users,
  projects,
  onAssignmentCreate,
  onAssignmentUpdate,
  onAssignmentDelete

  const [draggedAssignment, setDraggedAssignment] = useState<Assignment | null>(null)

      title: `${assignment.user.name} - ${assignment.project.name}`,
      start: new Date(assignment.start_date),
      end: new Date(assignment.end_date),
      resource: assignment.user.id,
      backgroundColor: getAssignmentColor(assignment),
      borderColor: assignment.conflicts?.length > 0 ? '#f44336' : undefined,
      extendedProps: {
        assignment,
        fte: assignment.fte_allocation,
        conflicts: assignment.conflicts || []
      }
    }))
  }, [assignments])
  
  const handleEventDrop = async (eventDropInfo: EventDropArg) => {
    const { event, delta } = eventDropInfo
    const assignment = event.extendedProps.assignment as Assignment
    
    const updatedAssignment = {
      ...assignment,
      start_date: event.start?.toISOString().split('T')[0],
      end_date: event.end?.toISOString().split('T')[0]
    }
    
    // Check for conflicts before confirming the move
    const conflicts = await detectAssignmentConflicts(updatedAssignment)
    if (conflicts.length > 0) {
      const confirmed = await showConflictConfirmation(conflicts)
      if (!confirmed) {
        eventDropInfo.revert()
        return
      }
    }
    
    await onAssignmentUpdate(assignment.id, {
      start_date: updatedAssignment.start_date,
      end_date: updatedAssignment.end_date
    })
  }
  
  const handleDateSelect = (selectInfo: DateSelectArg) => {
    // Open assignment creation dialog for selected date range
    openAssignmentDialog({
      start_date: selectInfo.start.toISOString().split('T')[0],
      end_date: selectInfo.end.toISOString().split('T')[0]
    })
  }
  
  return (
    <div className="assignment-calendar">

        view={view}
        onViewChange={setView}
        selectedDate={selectedDate}
        onDateChange={setSelectedDate}
      />

interface AssignmentSearchProps {
  onSearch: (filters: AssignmentFilters) => void
  onSavedSearchSelect: (savedSearch: SavedSearch) => void
}

const AssignmentSearch: React.FC<AssignmentSearchProps> = ({
  onSearch,
  onSavedSearchSelect
}) => {
  const [filters, setFilters] = useState<AssignmentFilters>({})
  const [savedSearches, setSavedSearches] = useState<SavedSearch[]>([])
  const [showAdvanced, setShowAdvanced] = useState(false)
  
  const handleFilterChange = (field: keyof AssignmentFilters, value: any) => {
    const newFilters = { ...filters, [field]: value }
    setFilters(newFilters)
    onSearch(newFilters)
  }
  
  const handleSaveSearch = async () => {
    const savedSearch = await saveAssignmentSearch({
      name: `Search ${Date.now()}`,
      filters: filters
    })
    setSavedSearches(prev => [...prev, savedSearch])
  }
  
  return (
    <Card className="assignment-search">
      <CardContent>
        <Grid container spacing={2}>
          <Grid item xs={12} md={4}>
            <TextField
              fullWidth
              label="Search assignments"
              placeholder="User name, project name, role..."
              value={filters.search || ''}
              onChange={(e) => handleFilterChange('search', e.target.value)}
              InputProps={{
                startAdornment: <SearchIcon />,
                endAdornment: filters.search && (
                  <IconButton onClick={() => handleFilterChange('search', '')}>
                    <ClearIcon />
                  </IconButton>
                )
              }}
            />
          </Grid>
          
          <Grid item xs={12} md={2}>
            <UserSelect
              label="User"

const useAssignmentError = () => {
  const [errors, setErrors] = useState<AssignmentError[]>([])
  
  const handleAssignmentError = useCallback((error: AssignmentError) => {
    setErrors(prev => [...prev, error])
    
    // Show appropriate user feedback based on error type
    switch (error.type) {
      case 'CAPACITY_EXCEEDED':
        showCapacityExceededDialog(error)
        break
      case 'ASSIGNMENT_CONFLICT':
        showConflictResolutionDialog(error)
        break
      case 'VALIDATION_ERROR':
        showValidationErrorSnackbar(error)
        break
      default:
        showGenericErrorSnackbar(error)
    }
  }, [])
  
  const clearError = useCallback((errorId: string) => {
    setErrors(prev => prev.filter(error => error.id !== errorId))
  }, [])
  
  return { errors, handleAssignmentError, clearError }
}

// Lazy loading and pagination
const useAssignmentPagination = (initialPageSize = 20) => {
  const [page, setPage] = useState(1)
  const [pageSize, setPageSize] = useState(initialPageSize)
  const [hasMore, setHasMore] = useState(true)
  
  const loadMore = useCallback(async () => {
    if (!hasMore) return
    
    const nextPage = page + 1
    const assignments = await loadAssignments({ page: nextPage, pageSize })
    
    if (assignments.length < pageSize) {
      setHasMore(false)
    }
    
    setPage(nextPage)
    return assignments
  }, [page, pageSize, hasMore])
  
  return { page, pageSize, hasMore, loadMore, setPage, setPageSize }
}
