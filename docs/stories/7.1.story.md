### FTE Validation and Capacity Logic
[Source: docs/prd/epic-7-team-assignment.md]
FTE allocation validation and capacity enforcement:
```python
class AssignmentFTEValidator:
    def validate_fte_allocation(self, fte: float) -> bool:
        """Validate individual assignment FTE allocation"""
        if not (0.1 <= fte <= 1.0):
            raise ValidationError("FTE allocation must be between 0.1 and 1.0")
        if round(fte, 1) != fte:
            raise ValidationError("FTE allocation must be in 0.1 increments")
        return True
    
    def validate_user_capacity(self, user_id: int, new_assignment: Assignment) -> bool:
        """Validate user total capacity across all assignments"""
        overlapping_assignments = self.get_overlapping_assignments(
            user_id, new_assignment.start_date, new_assignment.end_date
        )
        
        for date in self.get_date_range(new_assignment.start_date, new_assignment.end_date):
            total_fte = new_assignment.fte_allocation
            for assignment in overlapping_assignments:
                if self.date_in_range(date, assignment.start_date, assignment.end_date):
                    total_fte += assignment.fte_allocation
            
            if total_fte > 1.0:
                raise CapacityExceededError(
                    f"User capacity exceeded on {date}: {total_fte} FTE"
                )
        return True
```

### Conflict Detection and Resolution
Assignment conflict detection and classification:
```python
class AssignmentConflictDetector:
    def detect_conflicts(self, assignment: Assignment) -> List[AssignmentConflict]:
        """Detect all types of conflicts for an assignment"""
        conflicts = []
        
        # FTE overlap conflicts
        fte_conflicts = self.detect_fte_conflicts(assignment)
        conflicts.extend(fte_conflicts)
        
        # Date range conflicts
        date_conflicts = self.detect_date_conflicts(assignment)
        conflicts.extend(date_conflicts)
        
        # Role availability conflicts
        role_conflicts = self.detect_role_conflicts(assignment)
        conflicts.extend(role_conflicts)
        
        return conflicts
    
    def generate_resolution_suggestions(self, conflicts: List[AssignmentConflict]) -> List[ResolutionSuggestion]:
        """Generate resolution suggestions for detected conflicts"""
        suggestions = []
        
        for conflict in conflicts:
            if conflict.type == ConflictType.FTE_OVERLAP:
                suggestions.extend(self.suggest_fte_resolutions(conflict))
            elif conflict.type == ConflictType.DATE_OVERLAP:
                suggestions.extend(self.suggest_date_resolutions(conflict))
            elif conflict.type == ConflictType.ROLE_UNAVAILABLE:
                suggestions.extend(self.suggest_role_resolutions(conflict))
        
        return suggestions
```

### Capacity Constraint Enforcement
User capacity validation and enforcement:
```python
class CapacityConstraintEngine:
    def validate_assignment_capacity(self, assignment: Assignment) -> CapacityValidationResult:
        """Validate assignment against user capacity constraints"""
        user_assignments = self.get_user_assignments(assignment.user_id)
        capacity_timeline = self.build_capacity_timeline(user_assignments, assignment)
        
        violations = []
        for date, utilization in capacity_timeline.items():
            if utilization > 1.0:
                violations.append(CapacityViolation(
                    date=date,
                    utilization=utilization,
                    excess=utilization - 1.0
                ))
        
        return CapacityValidationResult(
            is_valid=len(violations) == 0,
            violations=violations,
            suggestions=self.generate_capacity_suggestions(violations, assignment)
        )
```
