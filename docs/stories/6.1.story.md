### FTE Allocation Validation Logic
FTE allocation validation and business rules:
```python
def validate_fte_allocation(fte_allocation: float, role_name: str) -> bool:
    """Validate individual role FTE allocation"""
    if not (0.1 <= fte_allocation <= 1.0):
        raise ValidationError("FTE allocation must be between 0.1 and 1.0")
    if fte_allocation % 0.1 != 0:
        raise ValidationError("FTE allocation must be in 0.1 increments")
    return True

def validate_total_sow_fte(sow_id: int) -> bool:
    """Validate total SOW FTE allocation across all roles"""
    total_fte = sum(role.fte_allocation for role in get_sow_roles(sow_id))
    if total_fte > 10.0:
        raise ValidationError("Total SOW FTE allocation cannot exceed 10.0")
    return True

def validate_sow_project_constraint(sow_id: int) -> bool:
    """Validate one active SOW per project constraint"""
    active_projects = get_active_projects_for_sow(sow_id)
    for project in active_projects:
        other_active_sows = get_other_active_sows_for_project(project.id, sow_id)
        if other_active_sows:
            raise ValidationError(f"Project {project.name} already has an active SOW")
    return True
```

- SOW status changes must consider linked project statuses
- Role FTE modifications must maintain total FTE constraints
- SOW date changes must not violate linked project date ranges
- One active SOW per project constraint must be enforced
- SOW client changes require validation of linked project clients

### SOW Cost Calculation and Budget Integration
SOW cost estimation and tracking:
```python
class SOWCostCalculator:
    def calculate_sow_cost(self, sow_id: int, duration_weeks: int) -> Dict:
        """Calculate total SOW cost based on role allocations and rates"""
        roles = self.get_sow_roles(sow_id)
        total_cost = 0
        role_costs = {}
        
        for role in roles:
            weekly_hours = role.fte_allocation * 40  # 40 hours per week
            total_hours = weekly_hours * duration_weeks
            role_cost = total_hours * role.hourly_rate
            role_costs[role.role_name] = {
                "hours": total_hours,
                "cost": role_cost,
                "fte": role.fte_allocation
            }
            total_cost += role_cost
        
        return {
            "total_cost": total_cost,
            "role_breakdown": role_costs,
            "duration_weeks": duration_weeks,
            "average_weekly_cost": total_cost / duration_weeks
        }
```
